截取指定字节数长度的字符串, 若最后的字占多个字节时则取全此字不截断, 此时实际字节长度大于指定字节长度

let sub = (s, l) => {
let c, r = '', t;
let i = 0
while(i < l) {
c = s.codePointAt(r.length);
c = c.toString(16);
i += Math.ceil(c.length / 2);
r += String.fromCodePoint('0x' + c);
}
return r;
}
console.log(sub('𠮷1𠮷1𠮷1𠮷1𠮷1𠮷1', 6));

let get_bytes = (s) => {
let c, r = '', t;
let i = 0;
while(r.length < s.length) {
c = s.codePointAt(r.length);
c = c.toString(16);
i += Math.ceil(c.length / 2);
r += String.fromCodePoint('0x' + c);
}
return i;
}
