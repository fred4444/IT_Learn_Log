//YYYY-MM-DD hh:mm:ss iiii
//i is milliseconds
//default pad is ' '
let get_date_str = function(date, format, pad) {
let year = date.getFullYear();
let month = date.getMonth() + 1;
let day = date.getDate();
let hour = date.getHours();
let minutes = date.getMinutes();
let seconds = date.getSeconds();
let milliseconds = date.getMilliseconds();

let m = [/[Y]+/g, /[M]+/g, /[D]+/g, /[h]+/g, /[m]+/g, /[s]+/g, /[i]+/g];
let p = [year, month, day, hour, minutes, seconds, milliseconds];
let a, s = format;
for (let i = 0, l = m.length; i < l; i++) {
a = s.match(m[i]);
if (a) {
a.forEach(function(v) {
let l = v.length;
let t = (p[i] + '').padStart(l, pad || ' ').substr(-l, l);
s = s.replace(v, t);
});
}
}
return s;
}

控制輸入框可輸入數字英文26個字母大寫及小寫還有中文

若只在input事件監聽函數中進行過濾會導致無法輸入中文, 因爲, 当使用中文输入法键入拼音时, 拼音字母已经写入输入框中, 触发了input事件, 但事实上用户并没有键入完毕, 所以, 当中文输入开始时，会触发compositionstart事件，此时做一个标记，告知程序用户正在中文输入，input事件触发时，判断当前是否正在键入中文，若是，则不作任何操作。
let input = document.getElementById('id_input');
let input_flag = false;
input.addEventListener('compositionstart', function(e) {
input_flag = true;
}, false);
input.addEventListener('compositionend', function(e) {
input_flag = false;
}, false);
input.addEventListener('input', function(e) {
if (input_flag) {
return;
}
let t = e.currentTarget;
t.value = t.value.replace(/[^0-9a-zA-Z\u4e00-\u9fa5]*/g, '');
}, false);

class Select_One_Component {
constructor(tag) {
this.tag = tag;
this.doc_obj = tag.ownerDocument;
}
get_seleted(d) {
let v, a, i = 0;
while (v = this.methods.get_next(d, i++)) {
if (this.methods.is_selected(v)) {
return v;
}
a = this.methods.get_children(v);
if (a) {
v = this.get_seleted(a);
if (v) {
return v;
}
}
}
return null;
}
set_all_selected(d, flag) {
let n, a, i = 0;
while (n = this.methods.get_next(d, i++)) {
this.methods.set_selected(n, flag);
a = this.methods.get_children(n);
if (a) {
this.set_all_selected(a);
}
}
}

//methods include : set_selected(v, flag), is_selected(v), get_next(d, i), get_children(v), get_selected_view(v, level)
set_data(data, methods, level = 0, deep = -1) {
this.data = data;
this.methods = methods;
this.level = level;
this.deep = deep;

this.visible_status = 0;//list_tag hide

this.tag.innerHTML = '';
let doc_obj = this.doc_obj;

this.show_tag = doc_obj.createElement('div');
let v = this.get_seleted(data);
if (!v) {
v = this.methods.get_next(data, 0);
}
this.selected_item = v;
let view;
if (v) {
let g_v = this.methods.get_selected_view || this.methods.get_view;
view = g_v(v, level);
if ('string' === typeof view) {
this.show_tag.innerHTML = view;
} else {
this.show_tag.appendChild(view);
}
} else {
this.show_tag.innerHTML = '';
}


this.list_tag = doc_obj.createElement('div');

this.click_lock = false;
this.list_tag.addEventListener('click', (e) => {
this.click_lock = false;
}, false);
this.list_tag.style.display = 'none';
this.tag.addEventListener('click', (e) => {
if (this.visible_status === 0) {
this.open(e);
} else {
this.close(e);
}
}, false);
this.draw();
}
draw(d = this.data, p_tag = this.list_tag, level = this.level, deep = this.deep) {

let v, view, c, n_level = level + 1, n_deep = deep - 1, ct, next_index = 0;
while (v = this.methods.get_next(d, next_index++)) {
view = this.methods.get_view(v, level);
if ('string' === typeof view) {
ct = document.createElement('div');
ct.innerHTML = view;
p_tag.appendChild(ct);
} else {
p_tag.appendChild(view);
ct = view;
}
ct.addEventListener('click', ((v) => {
return (e) => {
if (this.click_lock) {
return;
}
this.click_lock = true;
if (this.methods.change_selected) {
this.methods.change_selected(e, this.selected_item, v);
}
this.set_all_selected(d, false);
this.methods.set_selected(v, true);
let g_v = this.methods.get_selected_view || this.methods.get_view;
view = g_v(v, level);
this.show_tag.innerHTML = '';
if ('string' === typeof view) {
this.show_tag.innerHTML = view;
} else {
this.show_tag.appendChild(view);
}
};
})(v));
if ((c = this.methods.get_children(v)) && n_deep) {
this.draw(c, ct, n_level, n_deep);
}
}
this.tag.appendChild(this.show_tag);
this.tag.appendChild(this.list_tag);
}
close(option) {
let on = this.methods.onclose;
if (on) {
on(this, option);
}
this.list_tag.style.display = 'none';
this.visible_status = 0;
}
open(option) {
let on = this.methods.onopen;
if (on) {
on(this, option);
}
this.list_tag.style.display = 'block';
this.visible_status = 1;
}
get_current_item() {
return this.selected_item;
}
}
